59void actualiza_LED (uint16_t distancia_cm){
60 if (distancia_cm < 10){
61 LedOff(LED_1);
62 LedOff(LED_2);
63 LedOff(LED_3);
64 }
65 else if (distancia_cm >= 10 && distancia_cm < 20){
66 LedOn(LED_1);
67 }
68 else if (distancia_cm >=20 && distancia_cm < 30){
69 LedOn(LED_1);
70 LedOn(LED_2);
71 }
72 else if (distancia_cm >= 30){
73 LedOn(LED_1);
